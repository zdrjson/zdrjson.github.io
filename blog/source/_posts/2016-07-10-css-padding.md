# padding:10px 5px 15px 20px;


	•	上内边距是 10px
	•	右内边距是 5px
	•	下内边距是 15px
	•	左内边距是 20px
	
	
# padding:10px 5px 15px

  	•	上内边距是 10px
	•	右内边距和左内边距是 5px
	•	下内边距是 15px


# padding:10px 5px;

  	•	上内边距和下内边距是 10px
	•	右内边距和左内边距是 5px
	
	
# padding:10px;
  
	•	所有 4 个内边距都是 10px

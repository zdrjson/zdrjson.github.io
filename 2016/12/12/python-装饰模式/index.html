  • 不需要编写wrapper.name = func.name这样的代码，Python内置的functools.wraps就是
    干这个事的，所以，一个完整的decorator的写法如下：

def log(func):

@functools.wraps(func)

def wrapper(*args,**kw):

    print('call %s():' % func.__name__)

    return func(*args, **kw)

return wrapper

def log(text):

def decorator(func):

    @functools.wraps(func)

    def wrapper(*args, **kw):

        print('%s %s()' %(text, func.__name__))

        return func(*args, **kw)

    return wrapper

return decorator
